ข้อมูลถูกส่งไปเก็บที่ Google Sheets และ Drive เรียบร้อยแล้วครับ"); } catch (e) { console.error("Error: ", e); alert("❌ เกิดข้อผิดพลาดในการเชื่อมต่อ Google Drive: " + e.message); } finally { btn.disabled = false; btn.innerText = "บันทึกข้อมูลเข้าสวน"; status.style.display = 'none'; } }
